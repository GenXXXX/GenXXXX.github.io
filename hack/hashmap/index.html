        int sum = 0;
        int max = 0;
        
        //map stores the key-value pair of <sum of all elements from 0 to index i, i>

        for(int i = 0;i < nums.length;i++)
        {
            sum += nums[i];
            
            if(sum == k) max = i+1; //return the length of current array index
